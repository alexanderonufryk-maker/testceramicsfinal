    const questionText = document.getElementById('question-text');
    const answerButtonsElement = document.getElementById('answer-buttons');
    const nextBtn = document.getElementById('next-btn');
    const qCurrentDisplay = document.getElementById('q-current');
    const currentScoreDisplay = document.getElementById('current-score');
    const progressFill = document.getElementById('progress-fill');

    document.getElementById('start-btn').addEventListener('click', startQuiz);
    document.getElementById('restart-btn').addEventListener('click', startQuiz);
    
    nextBtn.addEventListener('click', () => {
        currentQuestionIndex++;
        if (currentQuestionIndex < currentQuestions.length) {
            setNextQuestion();
        } else {
            showResults();
        }
    });

    function startQuiz() {
        homeScreen.classList.add('hidden');
        resultScreen.classList.add('hidden');
        quizScreen.classList.remove('hidden');
        
        const shuffled = [...allQuestions].sort(() => 0.5 - Math.random());
        currentQuestions = shuffled.slice(0, 30);
        
        currentQuestionIndex = 0;
        score = 0;
        currentScoreDisplay.innerText = 0;
        setNextQuestion();
    }

    function setNextQuestion() {
        resetState();
        showQuestion(currentQuestions[currentQuestionIndex]);
        qCurrentDisplay.innerText = currentQuestionIndex + 1;
        
        const progress = ((currentQuestionIndex) / 30) * 100;
        progressFill.style.width = `${progress}%`;
    }

    function showQuestion(questionObj) {
        questionText.innerText = questionObj.q;
        const answers = [...questionObj.a].sort(() => 0.5 - Math.random());
        
        answers.forEach(answer => {
            const button = document.createElement('button');
            button.innerText = answer.t;
            button.classList.add('btn');
            if (answer.c) {
                button.dataset.correct = "true";
            }
            button.addEventListener('click', selectAnswer);
            answerButtonsElement.appendChild(button);
        });
    }

    function resetState() {
        nextBtn.disabled = true;
        while (answerButtonsElement.firstChild) {
            answerButtonsElement.removeChild(answerButtonsElement.firstChild);
        }
    }

    function selectAnswer(e) {
        const selectedBtn = e.target;
        const isCorrect = selectedBtn.dataset.correct === "true";
        
        if (isCorrect) {
            score++;
            currentScoreDisplay.innerText = score;
        }

        Array.from(answerButtonsElement.children).forEach(button => {
            setStatusClass(button, button.dataset.correct === "true");
            button.disabled = true; 
        });

        nextBtn.disabled = false;
        const progress = ((currentQuestionIndex + 1) / 30) * 100;
        progressFill.style.width = `${progress}%`;
    }

    function setStatusClass(element, correct) {
        element.classList.remove('correct');
        element.classList.remove('wrong');
        if (correct) {
            element.classList.add('correct');
        } else {
            element.classList.add('wrong');
        }
    }
